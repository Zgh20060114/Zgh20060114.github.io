基础：

verilog
module LED_Shift_Register (
    input clk,       // 时钟输入
    input rst,       // 复位信号
    input [7:0] sw,  // 拨码开关输入
    output reg [15:0] leds // LED输出
);

reg [3:0] counter; // 计数器

always @(posedge clk or posedge rst) begin
    if (rst) begin
        counter <= 4'b0000; // 复位计数器
    end else begin
        if (sw[7]) begin // 如果启停开关为1
            if (counter == 4'b1111) begin
                counter <= 4'b0000; // 当计数器到达最大值时，重置计数器
            end else begin
                counter <= counter + 1; // 否则递增计数器
            end
        end
    end
end

always @(posedge clk) begin
    if (sw[7]) begin // 如果启停开关为1
        if (sw[0]) begin // 如果SW0为1，从左向右
            leds <= {counter, 12'b0}; // 将计数器值作为LED输出
        end else begin // 如果SW0为0，从右向左
            leds <= {12'b0, counter}; // 将计数器值作为LED输出
        end
    end else begin
        leds <= 16'b0; // 否则关闭LED
    end
end

endmodule

自动换向：

verilog
module LED_Shift_Register_AutoDirection (
    input clk,       // 时钟输入
    input rst,       // 复位信号
    output reg [15:0] leds // LED输出
);

reg [3:0] counter; // 计数器
reg direction; // 方向信号，0表示向左，1表示向右

always @(posedge clk or posedge rst) begin
    if (rst) begin
        counter <= 4'b0000; // 复位计数器
        direction <= 1'b0; // 初始方向为向左
    end else begin
        if (counter == 4'b1111) begin
            counter <= 4'b0000; // 当计数器到达最大值时，重置计数器
            direction <= ~direction; // 切换方向
        end else begin
            counter <= counter + 1; // 递增计数器
        end
    end
end

always @(posedge clk) begin
    if (direction) begin // 如果方向为1，从右向左
        leds <= {12'b0, counter}; // 将计数器值作为LED输出
    end else begin // 如果方向为0，从左向右
        leds <= {counter, 12'b0}; // 将计数器值作为LED输出
    end
end

endmodule

进阶：

verilog
module LED_Shift_Register_AutoDirection_SpeedControl (
    input clk,       // 时钟输入
    input rst,       // 复位信号
    input sw0,       // 开关0，用于调节速度
    output reg [15:0] leds // LED输出
);

reg [3:0] counter; // 计数器
reg direction; // 方向信号，0表示向左，1表示向右
reg [3:0] speed; // 速度调节器

always @(posedge clk or posedge rst) begin
    if (rst) begin
        counter <= 4'b0000; // 复位计数器
        direction <= 1'b0; // 初始方向为向左
        speed <= 4'b0000; // 初始速度为最慢
    end else begin
        if (counter == speed) begin
            counter <= 4'b0000; // 当计数器计数达到速度值时，重置计数器
            if (sw0) begin // 如果SW0为1，增加速度
                if (speed < 4'b1111) begin
                    speed <= speed + 1; // 递增速度
                end
            end else begin // 如果SW0为0，减小速度
                if (speed > 4'b0000) begin
                    speed <= speed - 1; // 递减速度
                end
            end
            if (direction) begin // 如果方向为1，从右向左
                if (leds != 16'b0000000000000001) begin
                    leds <= {leds[13:0], 2'b00} // 向右移动
                end else begin
                    direction <= ~direction; // 切换方向
                end
            end else begin // 如果方向为0，从左向右
                if (leds != 16'b1000000000000000) begin
                    leds <= {2'b00, leds[15:2]}; // 向左移动
                end else begin
                    direction <= ~direction; // 切换方向
                end
            end
        end else begin
            counter <= counter + 1; // 递增计数器
        end
    end
end

endmodule
